# API's

status api =  urlpath:/node_status - method:get - res:{ status, node_id }
create bot = urlpath:/bot/add_bot - method:post - body:{ bot_token, bot_name, owner_id }
check node cap = urlpath:/max_cap' - method:get - res: { status, message }
get one bot data = urlpath:/bot/bot/:bot_id - method:get - res: { status ,bot }
get all my bots = urlpath:/bot/mybots/:uid - method:get - res:{ status, bots }
update bot data = urlpath:/bot/update - method:post - body:{ bot_id, bot_name, bot_token, st_message }
start a bot = urlpath:/bot/start - method:post - body: { bot_token }
bot status = urlpath:/bot/status/:token - method:get - res:{ status, message }
bot stop = urlpath:/bot/stop/:token - method:get - res:{ status, message }
delete bot = urlpath:/bot/delete - method:post - body: { bot_id }
get my servers = urlpath:/bot/servers/:token - method:get res: { status, message, server }
get channels = urlpath:/bot/channels - method:post body: { server_id, bot_token }
