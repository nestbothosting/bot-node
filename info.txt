# API's

status api =  urlpath:/node_status - method:get - res:{ status, node_id }
create bot = urlpath:/bot/add_bot - method:post - body:{ bot_token, bot_name, owner_id }
check node cap = urlpath:/max_cap' - method:get - res: { status, message }
get one bot data = urlpath:/bot/bot/:bot_id - method:get - res: { status ,bot }
get all my bots = urlpath:/bot/mybots/:uid - method:get - res:{ status, bots }
update bot data = urlpath:/bot/update - method:post - body:{ bot_id, bot_name, bot_token, st_message }
start a bot = urlpath:/bot/start - method:post - body: { bot_token }
bot status = urlpath:/bot/status/:token - method:get - res:{ status, message }
bot stop = urlpath:/bot/stop/:token - method:get - res:{ status, message }
delete bot = urlpath:/bot/delete - method:post - body: { bot_id }
get my servers = urlpath:/bot/servers/:token - method:get res: { status, message, server }
get channels = urlpath:/bot/channels - method:post body: { server_id, bot_token }
get roles = urlpath:/bot/roles - method:post body: { server_id, bot_token }
send Ticket =  urlpath:/event/ticket - method:post - body:{ ticketdata, fieldvalue, permission, bot_token }
send Embed message =  urlpath:/event/embed - method:post - body: { bot_token, fields, embed }
get Ticket Data = urlpath:/inticket/:server_id - method:get res: { status, ticketpanel }
delete ticket panel = urlpath:/delete_ticket_panel/:server_id - method:get - res: { status, message }

save YNS data = urlpath:/yns - method:post - body: { data:{}, token }
get YNS data urlpath:/yns_data/:server_id/:token - method:get - res : { status, message, channel_id, server_id, running, channel_name, channel_icon}
start YNS = urlpath:/start_listen_yns - method:POST - body : { bot_token, server_id }
stop YNS = urlpath:/stop_listen_yns - method:POST - body : { server_id }
delete YNS = urlpath:/delete_yns - method:POST - body : { server_id }
bot info (botstatus - page)  - urlpath:/bot/info/:bot_id - method:Get - res : { node_id, bot_name, node_id, status, bot_status }

Create Timedmessage - urlpath:/event/timedmessage - method:post - body : { status, message }
Get Timedmessage = urlpath:/event/my_timedmessage?server_id - method:get - res : { isTMS, TMS, status, message }
Start Timedmessage = urlpath:/event/start_timedmessage/:c_id - method:get - res : { status, message }
stop Timedmessage = urlpath:/event/stop_timedmessage/:c_id - method:get - res : { status, message }
Delete Timedmessage = urlpath:/event/delete_timedmessage - method:post - body : { c_id }
